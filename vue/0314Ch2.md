### fail
```
<!DOCTYPE html>
<html lang="en">
 <head>
	<script src="https://unpkg.com/vue@next"></script>
 </head>
 <body>
   <div id="app"> 
		<p> 總金額共 {{ subtotal() }} 元</p>
		<!--...中間略...-->
		<p> 總金額共 {{ subtotal() }} 元</p>
		<!--...中間略...-->
		<p> 總金額共 {{ subtotal() }} 元</p>
	</div>
	<script>
      const vm = Vue.creatApp({
		data () {
			return {
			    price:100,
				quality:10
			}
	},
	methods:{
			subtotal:function() {
				return this.price * this.quality;
				}
			}
	}).mount('#app');
   </script>   
 </body>    	
</html>
```
```
<!DOCTYPE html>
<html lang="en">
 <head>
 </head>
 <body>
   <div id="app"> 
		<p> 總金額共 {{ subtotalComputed }} 元</p>
		<p> 總金額共 {{ subtotalComputed }} 元</p>
		<p> 總金額共 {{ subtotalComputed }} 元</p>
		
		<p> 總金額共 {{ subtotalComputed }} 元</p>
		<p> 總金額共 {{ subtotalComputed }} 元</p>
		<p> 總金額共 {{ subtotalComputed }} 元</p>
	</div>
	<script src="https://unpkg.com/vue@next"></script>
	<script>
      const vm = Vue.creatApp({
		data () {
			return {
			    quantity:100,
				price:100
			}
	},
	computed: {
		subtotalComputed: function () {
			console.log('computed');
			return this.quantity * this.price;
		}
	},
	methods:{
			subtotalComputed: function () {
			console.log('methods');
			return this.quantity * this.price;
		}
	}).mount('#app');
   </script>   
 </body>    	
</html>
```
